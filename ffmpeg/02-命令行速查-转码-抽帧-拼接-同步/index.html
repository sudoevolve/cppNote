    02-е‘Ҫд»ӨиЎҢйҖҹжҹҘпјҡиҪ¬з ҒгҖҒжҠҪеё§гҖҒжӢјжҺҘгҖҒйҹіи§Ҷйў‘еҗҢжӯҘ¶

    зӣ®еҪ•йЎөпјҡи§Ғ FFmpeg ж–ҮжЎЈзӣ®еҪ•

    иҝҷз« еҸӘиҝҪжұӮвҖңиғҪзӣҙжҺҘжҠ„иө°з”ЁвҖқгҖӮе»әи®®е…Ҳз”Ё ffprobe зңӢжё…жҘҡе®№еҷЁ/зј–з Ғ/ж—¶й•ҝ/её§зҺҮ/йҹіиҪЁ/еӯ—е№•пјҢеҶҚеҶіе®ҡиҰҒдёҚиҰҒиҪ¬з ҒгҖӮ

    еҰӮжһңдҪ еҸ‘зҺ°иҮӘе·ұвҖңдјҡжҠ„дҪҶдёҚжҮӮвҖқпјҢе…ҲжҠҠиҝҷдёӨз« зңӢдёҖйҒҚпјҡ

    • жҰӮеҝөпјҡи§Ғ 04-жҰӮеҝөжҖ»и§Ҳпјҡе®№еҷЁгҖҒзј–з ҒгҖҒжөҒгҖҒж—¶й—ҙеҹәгҖҒеғҸзҙ ж јејҸ
    • еҸӮж•°пјҡи§Ғ 05-зј–з ҒеҸӮж•°дёҺиҙЁйҮҸпјҡCRFгҖҒз ҒзҺҮгҖҒйў„и®ҫгҖҒGOPгҖҒеғҸзҙ ж јејҸ

    1. е…ҲзңӢдҝЎжҒҜпјҡffprobe¶

    зңӢжөҒдҝЎжҒҜпјҲи§Ҷйў‘/йҹійў‘еҲҶеҲ«жҳҜд»Җд№Ҳзј–з Ғпјүпјҡ

    ffprobe -hide_banner -i input.mp4

    еҸӘзңӢж—¶й•ҝдёҺе…ій”®е…ғдҝЎжҒҜпјҲиҫ“еҮәжӣҙе№ІеҮҖпјүпјҡ

    ffprobe -v error -show_entries format=duration:format_tags=title -of default=nk=1:nw=1 input.mp4

    2. иҪ¬е°ҒиЈ…пјҲдёҚиҪ¬з ҒпјҢжңҖеҝ«пјү¶

    еҸӘжҚўе®№еҷЁпјҲдёҚйҮҚж–°зј–з Ғпјүпјҡ

    ffmpeg -i input.mp4 -c copy output.mkv

    еҸӘиҰҒи§Ҷйў‘пјҢдёҚиҰҒйҹійў‘/еӯ—е№•пјҡ

    ffmpeg -i input.mkv -map 0:v:0 -c copy out.mp4

    еҸӘиҰҒи§Ҷйў‘ + з¬¬дёҖжқЎйҹійў‘пјҲеӨҡйҹіиҪЁж–Үд»¶е»әи®®жҳҫејҸ mapпјүпјҡ

    ffmpeg -i input.mkv -map 0:v:0 -map 0:a:0 -c copy out.mkv

    3. еёёи§ҒиҪ¬з ҒпјҲйҖҡз”ЁжЁЎжқҝпјү¶

    иҪ¬ H.264 + AACпјҲжңҖйҖҡз”Ёзҡ„з»„еҗҲд№ӢдёҖпјүпјҡ

    ffmpeg -i input.mov -c:v libx264 -pix_fmt yuv420p -c:a aac -b:a 192k output.mp4

    зј©ж”ҫеҲ° 720pпјҲдҝқжҢҒе®Ҫй«ҳжҜ”пјүпјҡ

    ffmpeg -i input.mp4 -vf "scale=-2:720" -c:v libx264 -c:a copy output_720p.mp4

    4. жҲӘеҸ–/иЈҒеүӘпјҲе°ҪйҮҸдёҚиё©еҗҢжӯҘеқ‘пјү¶

    д»Һ 10s ејҖе§ӢеҸ– 5sпјҲжңҖзЁізҡ„ж–№ејҸпјҡйҮҚзј–з ҒдёҖйҒҚпјүпјҡ

    ffmpeg -ss 10 -t 5 -i input.mp4 -c:v libx264 -c:a aac out.mp4

    еҰӮжһңдҪ еҸӘжғівҖңж— жҚҹиЈҒеүӘвҖқпјҲ-c copyпјүпјҢйңҖиҰҒжҺҘеҸ—пјҡдёҚдёҖе®ҡд»ҺзІҫзЎ®её§еҲҮгҖҒд№ҹеҸҜиғҪйҹіз”»жңүеҒҸе·®пјҡ

    ffmpeg -ss 10 -t 5 -i input.mp4 -c copy out_copy.mp4

    5. жҠҪеё§/еҜјеҮәеӣҫзүҮеәҸеҲ—¶

    жҢүеё§зҺҮжҠҪпјҲжҜҸз§’ 2 её§пјүпјҡ

    ffmpeg -i input.mp4 -vf "fps=2" frames/%06d.png

    жҢүж—¶й—ҙжҲіжҠҪдёҖеј пјҲз¬¬ 3.5 з§’пјүпјҡ

    ffmpeg -ss 3.5 -i input.mp4 -frames:v 1 out.png

    6. еӣҫзүҮеәҸеҲ— вҶ” и§Ҷйў‘¶

    еӣҫзүҮеәҸеҲ—иҪ¬и§Ҷйў‘пјҲ30fpsпјүпјҡ

    ffmpeg -framerate 30 -i frames/%06d.png -c:v libx264 -pix_fmt yuv420p out.mp4

    и§Ҷйў‘иҪ¬еӣҫзүҮеәҸеҲ—пјҡ

    ffmpeg -i input.mp4 frames/%06d.png

    7. еҜјеҮәйҹійў‘ / ж”№ж јејҸ¶

    еҜјеҮә mp3пјҡ

    ffmpeg -i input.mp4 -vn -c:a libmp3lame -b:a 192k out.mp3

    еҜјеҮә wavпјҲPCMпјҢдҫҝдәҺеҒҡеҲҶжһҗ/жіўеҪўеӨ„зҗҶпјүпјҡ

    ffmpeg -i input.mp4 -vn -c:a pcm_s16le -ar 48000 -ac 2 out.wav

    8. еӯ—е№•пјҡе°ҒиЈ…дёҺзғ§еҪ•¶

    жҠҠ srt дҪңдёәиҪҜеӯ—е№•е°ҒиЈ…иҝӣ mkvпјҡ

    ffmpeg -i input.mp4 -i sub.srt -c:v copy -c:a copy -c:s srt out.mkv

    жҠҠеӯ—е№•зғ§иҝӣз”»йқўпјҲзЎ¬еӯ—е№•пјүпјҡ

    ffmpeg -i input.mp4 -vf "subtitles=sub.srt" -c:v libx264 -crf 20 -c:a copy out_hardsub.mp4

    9. ж—ӢиҪ¬/зҝ»иҪ¬¶

    ж—ӢиҪ¬ 90В°пјҡ

    ffmpeg -i input.mp4 -vf "transpose=1" -c:v libx264 -crf 20 -c:a copy out_rot.mp4

    ж°ҙе№ізҝ»иҪ¬пјҡ

    ffmpeg -i input.mp4 -vf "hflip" -c:v libx264 -crf 20 -c:a copy out_hflip.mp4

    10. ж°ҙеҚ°/еҸ еҠ ¶

    еҸ еҠ  logo еҲ°еҸідёҠи§’пјҡ

    ffmpeg -i input.mp4 -i logo.png -filter_complex "overlay=W-w-20:20" -c:v libx264 -crf 20 -c:a copy out_logo.mp4

    11. еёёи§ҒиҙЁйҮҸ/дҪ“з§ҜжҺ§еҲ¶¶

    CRF + presetпјҲзҰ»зәҝеҺӢзј©еёёз”Ёпјүпјҡ

    ffmpeg -i input.mp4 -c:v libx264 -crf 20 -preset slow -pix_fmt yuv420p -c:a aac -b:a 160k out.mp4

    ејәиЎҢеҚЎз ҒзҺҮпјҲеёҰе®ҪеҸ—йҷҗ/жҺЁжөҒеёёз”Ёпјүпјҡ

    ffmpeg -i input.mp4 -c:v libx264 -b:v 2500k -maxrate 2500k -bufsize 5000k -g 60 -pix_fmt yuv420p -c:a aac -b:a 160k out.mp4

    12. жӢјжҺҘпјҲconcatпјү¶

    еҗҢзј–з ҒеҸӮж•°зҡ„ mp4 жӢјжҺҘпјҲжҺЁиҚҗе…ҲиҪ¬е°ҒиЈ…еҲ° TS еҶҚ concatпјҢе…је®№жҖ§жӣҙеҘҪпјүпјҡ

    ffmpeg -i a.mp4 -c copy -bsf:v h264_mp4toannexb -f mpegts a.ts
ffmpeg -i b.mp4 -c copy -bsf:v h264_mp4toannexb -f mpegts b.ts
ffmpeg -i "concat:a.ts|b.ts" -c copy -bsf:a aac_adtstoasc out.mp4

    еҰӮжһңдҪ иғҪдҝқиҜҒиҫ“е…Ҙж–Үд»¶зј–з ҒеҸӮж•°е®Ңе…ЁдёҖиҮҙпјҢд№ҹеҸҜд»Ҙз”Ё concat demuxerпјҲжӣҙж ҮеҮҶпјүпјҡ

    еҮҶеӨҮдёҖдёӘ list.txtпјҡ

    file 'a.mp4'
file 'b.mp4'

    з„¶еҗҺпјҡ

    ffmpeg -f concat -safe 0 -i list.txt -c copy out.mp4

    13. еёёи§ҒеҗҢжӯҘй—®йўҳзҡ„з¬¬дёҖеҸҚеә”¶

    • е…Ҳз”Ё ffprobe зңӢпјҡжҳҜеҗҰ VFRпјҲеҸҜеҸҳеё§зҺҮпјүгҖҒйҹійў‘йҮҮж ·зҺҮжҳҜеҗҰдёҖиҮҙгҖҒжҳҜеҗҰжңүеӨҡйҹіиҪЁ
    • еҸӘиҰҒдҪ дёҚжҳҜејәеҲ¶ж— жҚҹпјҲ-c copyпјүпјҢеӨҡж•°еҗҢжӯҘй—®йўҳйғҪиғҪйҖҡиҝҮвҖңйҮҚзј–з Ғ + з»ҹдёҖеҸӮж•°вҖқи§ЈеҶі

    еҰӮжһңдҪ з»ҸеёёйҒҮеҲ°вҖңйҖүй”ҷйҹіиҪЁ/еӯ—е№•/иҺ«еҗҚе…¶еҰҷдёҚеҗҢжӯҘ/жҠҘй”ҷзңӢдёҚжҮӮвҖқпјҢзӣҙжҺҘзңӢпјҡ

    • 07-ffprobe дёҺжҺ’й”ҷпјҡжөҒйҖүжӢ©пјҲmapпјүгҖҒеҗҢжӯҘгҖҒеёёи§ҒжҠҘй”ҷ
